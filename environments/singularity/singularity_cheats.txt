
# open a shell from a docker image:
singularity shell --nv docker://ffdanff/pytorch-oiio:latest

# if done repeatedly, does this use the singularity image?  or re-convert the docker one?

# check how much data singularity uses
singularity cache list

# execute something in a singularity container
singularity exec --nv docker://tensorflow/tensorflow:latest-gpu python ./models/tutorials/image/mnist/convolutional.py

# use spython (requires separate install) to convert a dockerfile to a singularity recipe
spython recipe Dockerfile Singularity

# build a recipe
singularity build nuke_ml_server.sif Singularity

# build a recipe with a sandbox:
singularity build --sandbox sandbox/  nuke_ml_server.sif Singularity

# shell with access to host filesystem
singularity shell --nv --bind /mnt/ala nst.sif

# shell with access to host filesystem and sandbox writable:
singularity shell --writable sandbox/ --nv --bind /mnt/ala nst.sif
